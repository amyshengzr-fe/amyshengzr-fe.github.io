  let left = 0;
  let right = nums.length - 1;

  while (left <= right) {
    const middle = Math.floor((left + right) / 2);
    const num = nums[middle];

    if (num === target) {
      return middle;
    } else if (num > target) {
      right = middle - 1;

      left = middle + 1;

  }

  return -1;
